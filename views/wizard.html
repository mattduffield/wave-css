wc-wizard

Inline Wizard

Step 1: Project Basics

Step 2: Technology Stack

Step 3: Review

Review your selections and click Save.

 

      


      
      

        
Public API

        

          wizard.next() — Go to next step

          wizard.back() — Go to previous step

          wizard.goTo(n) — Go to step by index (0-based)

          wizard.getCurrentStep() — Current step index

          wizard.getTotalSteps() — Total number of steps

          wizard.isFirst() / wizard.isLast() — Position checks
        

      


      
      

        
Usage

        
<wc-wizard id="my-wizard">
  <wc-wizard-step label="Basics">
    <wc-input name="name" lbl-label="Name"></wc-input>
  </wc-wizard-step>
  <wc-wizard-step label="Details" before-navigate="saveState">
    <wc-select name="type" lbl-label="Type"></wc-select>
  </wc-wizard-step>
  <wc-wizard-step label="Preview">
    <!-- Lazy-load via HTMX if needed -->
    <div hx-get="/api/preview" hx-trigger="load"></div>
  </wc-wizard-step>
</wc-wizard>

      


    

  


  
  
    window.demoBeforeNavigate = function() {
      var log = document.getElementById('demo-log');
      if (log) log.textContent += 'before-navigate called on Tech Stack step\n';
    };

    var wizard = document.getElementById('demo-wizard');
    if (wizard) {
      wizard.addEventListener('wcwizardstepchange', function(e) {
        var d = e.detail;
        var log = document.getElementById('demo-log');
        if (log) log.textContent = 'step: ' + d.step + ' / ' + d.totalSteps + ' | isFirst: ' + d.isFirst + ' | isLast: ' + d.isLast;

        document.getElementById('demo-back').style.display = d.isFirst ? 'none' : '';
        document.getElementById('demo-next').style.display = d.isLast ? 'none' : '';
        document.getElementById('demo-save').style.display = d.isLast ? '' : 'none';
      });
    }
