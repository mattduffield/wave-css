Usage Examples

<!-- Basic icon -->
<wc-icon name="house" icon-style="solid"></wc-icon>

<!-- Icon with size and color -->
<wc-icon name="heart" icon-style="solid" size="2rem" color="#ef4444"></wc-icon>

<!-- Rotated icon -->
<wc-icon name="arrow-right" icon-style="solid" rotate="90"></wc-icon>

<!-- Duotone icon -->
<wc-icon name="camera" icon-style="duotone" 
  primary-color="#3b82f6" 
  secondary-color="#93c5fd"></wc-icon>
