Usage Examples

<!-- Basic icon with utility classes -->
<wc-icon name="house" icon-style="solid" class="w-5 h-5"></wc-icon>

<!-- Icon with color utilities -->
<wc-icon name="heart" icon-style="solid" class="w-6 h-6 text-red-500"></wc-icon>

<!-- Inline with text -->
<span class="flex items-center gap-2">
  <wc-icon name="check" icon-style="solid" class="w-4 h-4 text-green-500"></wc-icon>
  Complete
</span>

<!-- Centered in a button -->
<button class="flex items-center gap-2 px-4 py-2 bg-blue-500 text-white rounded">
  <wc-icon name="download" icon-style="solid" class="w-4 h-4"></wc-icon>
  Download
</button>
